The model contract

  • get() reads from the store or from a specific todo item
  • set(value) writes back through store.set() or an action
  • signal decides when the DOM control should resync
  • event and prop let the binding target text inputs, checkboxes and selects
 

export function todoModel(todoId, field, options = {}) {
  return model({
    signal: mainState,
    get: () => getTodoById(todoId)?.[field]
      ?? (options.prop === 'checked' ? false : ''),
    set: (value) => updateTodo(todoId, { [field]: value }),
    ...options,
  })
}
