helpers/_helpers.js: two tiny translators for all our inputs

We do not want every text field and checkbox to remember how to read and write the store. So the app builds two small helpers, and then reuses them everywhere.
  • storeModel(path) is the generic translator for fields like filters.search or preferences.theme
  • todoModel(todoId, field) is the special translator for one field on one todo card
  • Both helpers are just model({ signal, get, set }), which means the same directive powers the whole app
  • This is the practical use of the template engine chapter: tiny reusable bindings instead of manual event code everywhere
 

export function storeModel(path, options = {}) {
  return model({
    signal: tickState,
    get: () => store.get(path),
    set: (value) => store.set(path, value),
    ...options,
  })
}

export function todoModel(todoId, field, options = {}) {
  return model({
    signal: tickState,
    get: () => getTodoById(todoId)?.[field],
    set: (value) => updateTodo(todoId, { [field]: value }),
    ...options,
  })
}
