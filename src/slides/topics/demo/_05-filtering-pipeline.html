data/_pipeline.js: one filter pipe, one job at a time

Instead of making lots of temporary arrays, the app lets todos walk through a pipe. Each filter looks at one todo, decides yes or no, and passes only the matching ones to the next step.
  • fromArray() turns the todo list into a lazy stream
  • filterBySearch(), filterByCategory(), filterByStatus(), and filterByPriority() each do one simple job
  • Only the final sort step materializes the list, so the pipeline stays cheap and easy to inspect
  • visibleTodos simply asks the pipeline for the ready-made answer
 

export function pipelineTodos(todos, filters, language) {
  const iterator = filterByPriority(
    filterByStatus(
      filterByCategory(
        filterBySearch(fromArray(todos), filters.search, language),
        filters.category,
      ),
      filters.status,
    ),
    filters.priority,
  )

  return sortTodos(iterator, filters.sortBy, filters.sortDir, language)
}
