state/_actions.js: the hands that move the toys

Actions are the places where the app says "change the world now". They are boring on purpose: find something, replace it, add it, remove it, or rebuild the whole state from the seed.
  • updateTodo() finds one item and replaces it with a patched copy
  • addTodo() turns the shared draft into a new todo, then resets the draft for the next creation
  • deleteSelected(), deleteCompleted(), and friends operate on whole arrays with plain map() or filter()
  • resetDemo() replaces the entire state tree while keeping the current preferences
 

export function updateTodo(id, patch) {
  const index = store.state.todos.findIndex((todo) => todo.id === id)
  if (index < 0) return
  const current = store.state.todos[index]
  store.state.todos[index] = {
    ...current,
    ...patch,
  }
}

export function resetDemo() {
  const { preferences } = store.snapshot()
  store.replace({
    ...createSeedData(),
    preferences: { ...preferences },
  })
  tickState.set(performance.now())
}
