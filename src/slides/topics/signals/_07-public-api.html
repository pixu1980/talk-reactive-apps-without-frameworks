index.js: the small public surface the rest of the app actually imports

Internal files are easier to teach when they stay tiny and focused. The outside world, though, wants one friendly door. This module assembles that door: constructors, one escape hatch for untracked reads, one type guard, and one helper to create effects.
  • Signal.State and Signal.Computed expose the two main reactive primitives without leaking file-level implementation details
  • Signal.subtle.untrack intentionally sounds advanced, because turning tracking off is powerful and should feel explicit
  • isSignalLike() gives the renderer a duck-typing check instead of forcing a brittle instanceof contract
  • effect() creates an EffectCollector and returns the disposer, which makes lifecycle cleanup natural for callers
 

export const Signal = {
  State: StateSignal,
  Computed: ComputedSignal,
  subtle: {
    untrack: withoutCollector,
  },
}

export function isSignalLike(value) {
  return Boolean(
    value &&
    typeof value.get === "function" &&
    value.__isSignal
  )
}

export function effect(callback) {
  const runner = new EffectCollector(callback)
  return () => runner.stop()
}
