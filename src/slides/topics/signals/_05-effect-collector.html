_effect-collector.js: the runner that lets the graph touch the outside world

Effects are where reactivity becomes visible. Rendering, logging, syncing, measuring: none of those produce a derived value, so they live in a collector that tracks reads, resubscribes on every run, and hands reruns to the scheduler.
  • The constructor initializes the callback, dependency map, and active flag, then immediately runs once to discover the first dependency graph
  • addDependency() subscribes to each signal with () => schedule(this), which means invalidations are batched instead of running effects synchronously
  • run() first cleans stale subscriptions, then executes the callback inside withCollector(this, ...) so fresh reads rebuild the graph
  • stop() is the disposer: it flips active off and unsubscribes from everything
 

export class EffectCollector {
  constructor(callback) {
    this.callback = callback
    this.dependencies = new Map()
    this.active = true
    this.run = this.run.bind(this)
    this.run()
  }

  addDependency(signal) {
    if (this.dependencies.has(signal)) return
    const unsubscribe = signal.subscribe(() => schedule(this))
    this.dependencies.set(signal, unsubscribe)
  }

  run() {
    if (!this.active) return
    this.cleanup()
    withCollector(this, () => {
      this.callback()
    })
  }
}
