_state-signal.js: the box that actually holds a value

If BaseSignal is the wiring, StateSignal is the first real piece of state. It stores one concrete value, exposes tracked and untracked reads, and only notifies dependents when the value really changes.
  • It extends BaseSignal, so it inherits subscription, tracking, and notification behavior instead of reimplementing them
  • get() calls track() before returning the value, which is how reads become graph edges
  • peek() returns the same value without dependency collection, useful when you want to inspect state quietly
  • set() uses equals as a guard rail, so identical writes do not fan out pointless invalidations
 

export class StateSignal extends BaseSignal {
  constructor(value, options = {}) {
    super()
    this.value = value
    this.equals = options.equals ?? Object.is
  }

  get() {
    this.track()
    return this.value
  }

  peek() {
    return this.value
  }

  set(nextValue) {
    if (this.equals(this.value, nextValue)) return this.value
    this.value = nextValue
    this.notify()
    return this.value
  }
}
