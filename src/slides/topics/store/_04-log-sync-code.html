window.addEventListener('store:change', (event) => {
  if (isWritingDebugLog) return

  if (!store.state.debug.paused &&
      event.detail.path !== 'debug.logs') {
    isWritingDebugLog = true
    try {
      store.state.debug.logs = [
        {
          id: crypto.randomUUID(),
          timestamp: new Date().toLocaleTimeString('en-GB'),
          ...event.detail,
        },
        ...store.state.debug.logs,
      ].slice(0, 30)
    } finally {
      isWritingDebugLog = false
    }
  }

  localStorage.setItem(STORAGE_KEY,
    JSON.stringify(store.snapshot()))
  mainState.set(performance.now())
})
