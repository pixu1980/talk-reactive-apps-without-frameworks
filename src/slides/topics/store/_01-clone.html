_clone.js: the store's washing machine for values

Before the store keeps data, emits events, or returns snapshots, it runs values through one careful cloning pipeline. This file exists so the outside world never gets to hold the store's raw internals by accident.
  • getRawValue() unwraps proxies through __raw, so cloning works on the actual target object instead of proxy wrappers
  • clonePlainValue() uses a WeakMap named seen to preserve identity across repeated references and avoid infinite loops
  • It explicitly preserves Date, RegExp, Map, Set, arrays, and plain enumerable object fields
  • deepClone() is just the friendly public wrapper the rest of the store calls when it needs an immutable snapshot
 

function getRawValue(value) {
  if (!isObject(value)) return value
  return value.__raw ?? value
}

export function clonePlainValue(value, seen = new WeakMap()) {
  const raw = getRawValue(value)
  if (!isObject(raw)) return raw
  if (seen.has(raw)) return seen.get(raw)

  if (raw instanceof Date) return new Date(raw.getTime())
  if (raw instanceof RegExp) return new RegExp(raw.source, raw.flags)
  if (raw instanceof Map) {
    const next = new Map()
    seen.set(raw, next)
    for (const [key, entryValue] of raw.entries()) {
      next.set(clonePlainValue(key, seen), clonePlainValue(entryValue, seen))
    }
    return next
  }
}
