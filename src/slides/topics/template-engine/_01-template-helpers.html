_template-helpers.js: the tiny IR every template expression becomes

Before the engine can render anything, it needs one honest shape for "a template", and one honest shape for "a directive". This file gives us exactly that, with almost no magic.
  • html(strings, ...values) does not touch the DOM; it returns a plain object with kind, strings, and values
  • directive(name, payload) brands directive results with a private Symbol, so parts can recognize them without stringly typed hacks
  • isDirective() checks that brand and can optionally narrow by name, which is how parts distinguish model from repeat
  • model() and repeat() are just friendly wrappers over directive(), so the authoring API stays declarative
 

export function html(strings, ...values) {
  return {
    kind: "template-result",
    strings,
    values,
  }
}

export function directive(name, payload) {
  return {
    [directiveBrand]: true,
    name,
    payload,
  }
}

export function model(config) {
  return directive("model", config)
}

export function repeat(items, key, renderItem) {
  return directive("repeat", { items, key, renderItem })
}
