_property-part.js: write to DOM properties, not HTML text

Attributes are strings on markup. Properties are live JavaScript fields on elements. This part exists for the moments when the engine must talk to the actual DOM object, not to serialized HTML.
  • It extends Part, so property bindings automatically gain the same signal subscription and cleanup behavior as the other reactive parts
  • setValue() branches only once: if the input is signal like, subscribe and resolve it; otherwise commit the raw value
  • commit() writes straight to this.element[this.name], which is why booleans, objects, and live form values work correctly
  • The file is intentionally tiny because all the complexity belongs in the DOM, not in the wrapper around it
 

export class PropertyPart extends Part {
  constructor(element, name) {
    super()
    this.element = element
    this.name = name
  }

  setValue(value) {
    if (isSignalLike(value)) {
      this.bindSignal(value, (resolved) => this.commit(resolved))
      return
    }
    this.disposeSignal()
    this.commit(value)
  }

  commit(value) {
    this.element[this.name] = value
  }
}
