_template-instance.js: compile once, clone fast, bind precise parts

This is the parser and the materializer. It turns template literals into cached descriptors, then turns those descriptors into concrete part objects inside a cloned fragment. If the engine feels magical anywhere, it is here.
  • getTemplate() builds synthetic markup with placeholders, parses it once, walks the fragment, and stores descriptor metadata in a WeakMap keyed by the original TemplateStringsArray
  • Descriptors record type, index, name, raw attribute name, and node path, so later instances can find the exact DOM spot without reparsing strings
  • The constructor clones the cached template fragment, resolves descriptor paths, and swaps placeholders for ChildNodePart, AttributePart, PropertyPart, or EventPart
  • update(values) is then almost boring: walk the values array and push each value into its matching part
 

function getTemplate(strings) {
  let record = templateCache.get(strings)
  if (record) return record

  let markup = ""
  for (let index = 0; index < strings.length - 1; index += 1) {
    const chunk = strings[index]
    markup += chunk
    markup += chunk.match(ATTRIBUTE_PART_RE)
      ? `__part_${index}__`
      : ``
  }

  const template = document.createElement("template")
  template.innerHTML = markup + strings[strings.length - 1]
  record = { template, descriptors }
  templateCache.set(strings, record)
  return record
}

update(values) {
  for (let index = 0; index < values.length; index += 1) {
    const part = this.parts.get(index)
    if (part) part.setValue(values[index])
  }
}
