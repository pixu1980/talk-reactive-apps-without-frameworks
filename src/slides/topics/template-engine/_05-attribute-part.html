_attribute-part.js: plain attributes plus the most opinionated directive in the engine

This file starts simple and then becomes surprisingly sophisticated. It can write ordinary attributes, subscribe to signals, and run the two way model() directive with cursor preservation and a select specific bug fix.
  • setValue() is the traffic cop: route model(...) to commitModel(), route signals through bindSignal(), otherwise commit a normal attribute
  • commit() maps null and false to attribute removal, true to an empty attribute, and everything else to a string
  • commitModel() caches bindings by event, property, and signal identity, so stable model shapes do not tear down listeners every rerender
  • sync() handles checked versus value, preserves text selection while typing, and syncAfterRender() gives <select> one extra microtask so options exist first
 

setValue(value) {
  if (this.name === "model" && isDirective(value, "model")) {
    this.commitModel(value.payload)
    this.value = value
    return
  }

  if (isSignalLike(value)) {
    this.disposeModel()
    this.bindSignal(value, (resolved) => this.commit(resolved))
    return
  }

  this.disposeModel()
  this.disposeSignal()
  this.commit(value)
}

const syncAfterRender = () => {
  sync()
  if (this.element instanceof HTMLSelectElement) {
    queueMicrotask(() => {
      if (this._modelBinding === binding && this.element.isConnected) sync()
    })
  }
}
